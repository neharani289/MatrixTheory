\section{\textbf{Problem}}
Let 
\begin{align}
\vec{A} = \myvec{2&-1&1\\1&2&1},
\vec{B}= \myvec{3\\1\\-1},
\vec{C}=\myvec{1&-1}
\end{align}
Compute $\vec{ABC}$ and $\vec{CAB}$.
 \section{\textbf{Solution}} 
 %
 %
 $\vec{ABC}$ = ($\vec{AB})$ $\vec{C}$
 \begin{align}
 \vec{AB}= \myvec{2&-1&1\\1&2&1}\myvec{3\\1\\-1}\\
 \vec{AB}=\myvec{6-1-1\\3+2-1}\\
 \implies \vec{AB}=\myvec{4\\4}
 \end{align}
 \begin{align}
     \vec{ABC}= \myvec{4\\4} \myvec{1&-1}\\
    \implies \vec{ABC}= \myvec{4 & -4\\4 & -4}
 \end{align}
$\vec{CAB}$ = $\vec{C}$($\vec{AB}$) \begin{align}
 \vec{CAB} = \myvec{1&-1} \myvec{4&-4\\4&-4}\\
 \implies \vec{CAB} = \myvec{0}
 \end{align}
  therefore,
\begin{align}
    \vec{ABC}=\myvec{4&-4\\4&-4}\\
    \vec{CAB}=\myvec{0}
\end{align}
