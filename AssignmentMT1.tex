\noindent Q no. 46. what are the points on the y-axis whose distance from the line (4 3)x = 12 is 4 units.
\\
\solution 
\begin{enumerate}
\noindent Here ,direction vectors of the lines are
\myvec{3 \\ -4} \\
\noindent normal vector n is given by,
\end{enumerate}
%
\begin{align}
\ n&=\myvec{4\\3}
\end{align}
%

\begin{enumerate}
\noindent now distance between a given line which meet y-axis at point \vec{P_1} and  desired point\vec{P_2} and \vec{P_3}
\\
\end{enumerate}
%
\begin{align}
\ d &= \frac{(\vec{P_1}-\vec{P_2})^T n}{||n||} \\
\implies 4 &= \frac{\myvec{0 & \brak{4-a}}^T\myvec{4\\3}}{5}
\implies 20 &= 12-3a\\
\ a &= -8/3
\end{align}
%
\begin{enumerate}
\noindent similarly distance between given line and the desired point P3 can be calculated as:\\
\end{enumerate}
%
\begin{align}
\ d &= \frac{(\vec{P_3}-\vec{P_1})^T n}{||n||}\\
\implies 4 &= \frac{\myvec{0 & \brak{b-4}}^T\myvec{4\\3}}{5}\\
\ 20 &= 3b-12\\
\ b &= 32/3
\end{align}
\begin{enumerate}
\noindent therefore points on y-axis at 4 units distance from line are (0,-8) and (0,32/3).
\end{enumerate}
