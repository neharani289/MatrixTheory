\section{\textbf{Problem}}
(UGC-june2017,71) : \\
%
Let \textbf{V} be the vector space of polynomials of degree at most 3 in a varible x with coefficients in $\mathbb{R}$. Let \textbf{T}=d/dx be the linear transformation of \textbf{V} to itself given by differentiation.\\

Which of the following are correct?\\
\begin{enumerate}
\item $\vec{T}$ is invertible
\item 0 is an eigenvalue of $\vec{T}$
\item There is a basis with respect to which the matrix of \textbf{T} is nilpotent.
\item The matrix of \textbf{T} with respect to the basis \myvec{1,1+x,1+x+x^2,1+x+x^2+x^3} is diagonal.
\end{enumerate}
\section{\textbf{Solution}}
	\begin{longtable}{|l|l|}
		\hline
		\multirow{3}{*}{Checking whether  matrix $\vec{T}$ is nilpotent} &\\
	 &	Let \textbf{V}= P_3(x)\\

     & \textbf{T}:\textbf{V}\rightarrow \textbf{V}\\
     & \textbf{T}(P(x)) = P'(x)\\
     

		&Standard basis of P(x)=\brak{1,x,x^2,x^3}\\
		&for finding P'(x) , differentiating the standard  basis;\\
    	& \qquad \qquad \qquad 
       \vec{T}(1)=0=a_1x+b_1x+c_1x^2+d_1x^3\\
       & \qquad \qquad \qquad 
       \vec{T}(x)=1=a_2+b_2x+c_2x^2+d_2x^3\\
       & \qquad \qquad \qquad 
       \vec{T}(x^2)=2x=a_3+b_3x+c_3x^2+d_3x^3\\
        & \qquad \qquad \qquad 
        \vec{T}(x^3)=3x^2=a_4+b_4x+c_4x^2+d_4x^3\\
    & 
         Representing \textbf{T} in matrix form ;\\
    &
   \qquad\qquad\qquad \vec{T}=\myvec{0&1&0&0\\0&0&2&0\\0&0&0&3\\0&0&0&0}\\
   &
  from the above matrix \textbf{T} we can say it is nilpotent matrix.
    \hline
\end{longtable}

	\begin{longtable}{|l|l|}
	\hline
		\multirow{3}{*}{ Checking for eigen value of  $\vec{T}$} 
		&
		
		\vec{T}=\myvec{0-\lambda&1&0&0\\0&0-\lambda&2&0\\0&0&0-\lambda&3\\0&0&0&0-\lambda}\\
		&
		\implies \lambda=0\\
		&\\
	   	\hline
	   \multirow{3}{*}{ Checking whether $\vec{T}$ is invertible } & \\
	   &
	   Since \det{T}=0. \\
	   &
	   Therefore $\vec{T}$ is not invertible 
	   	\hline
	   	\multirow{3}{*}{Matrix $\vec{T}$
	   	is diagonal matrix} & \\
	   			& Let basis of P(x) be B'=\brak{1,1+x,1+x+x^2,1+x+x^2+x^3}\\
		& Differentiating wrt x ;\\
    	&
       \vec{T}(1)=0=a_1x+b_1(1+x)+c_1(1+x+x^2)+d_1(1+x+x^2+x^3)\\
       &
       \vec{T}(1+x)=1=a_2+b_2(1+x)+c_2(1+x+x^2)+d_2(1+x+x^2x^3)\\
       &
       \vec{T}(1+x+x^2)=1+2x=a_3+b_3(1+x)+c_3(1+x+x^2)+d_3(1+x+x^2+x^3)\\
        &  
        \vec{T}(1+x+x^2+x^3)=1+2x+3x^2=a_4+b_4(1+x)+c_4(1+x+x^2)\\
        &
     \qquad\qquad\qquad\qquad+d_4(1+x+x^2+x^3)\\
    &
    \qquad\qquad\qquad \vec{T}=\myvec{0&1&-1&-1\\0&0&2&-1\\0&0&0&3\\0&0&0&0}\\
   &
     above matrix is not a diagonal matrix
	   	\hline
	   	\multirow{3}{*}{Conclusion} & \\
	&
	Thus we can conclude \\
		&
		Option 2) and 3) are correct.\\
